        - req
@param {init}   - pid 省
@param {init}   - cityid 市
@param {string} - sex
@url post User/GetUserFinancialList
- res
@return success{
    data:{
        list:[
            {
                "RealName": '某某‘, - 真实姓名
                "authType":1       - 1通过名片认证, 2通过答题
                "capacity": null,  - 管理资产
                "city": 310100,    - 城市
                "count": 3         - 服务客户
            }
        ]
    },
    totalCount:2
}
@return failed{
    errMsg:'未找到您要查找的理财师！'
}
